@each $section, $colour in $section-colours {

		color: white;
		}

			@include button;
			border-color: $colour;

		}
	}
	.#{$section}:hover {
		background-color: $colour;

		}

This is an each loop I used in my HTML and CSS assessment to quickly and easily colour the various company divisions.
Coded in Sass, it loops through a map that contains the section name as the key, and the section colour as the value.

const navbar = document.getElementById("navbar");
const menuButton = document.getElementById("menu-btn");
let navActive = false;

menuButton.onclick = function() {show_navbar()};

function show_navbar() {
	if (navActive === false) {
		navbar.classList.add("nav-active");
		menuButton.style.animation = "button-spin-1 0.4s ease-in-out 0s 1 normal none";
		navActive = true;
	} else {
		navbar.classList.remove("nav-active");
		menuButton.style.animation = "button-spin-2 0.4s ease-in-out 0s 1 reverse none";
		navActive = false;
		setTimeout(() => {
			menuButton.style.animation = "logo-pulse 1.5s ease-in-out 0s infinite normal both"
		}, 400)
	}
};

This is a Javascript function I used in this very website to animate the logo and navigation bar transition.
Coded in pure Javascript, it listens for a click or touchscreen press on the logo, adds or removes an HTML class to the navbar element - applying or removing pre-written CSS styling in the process, and animates the logo with either the "spin" transition animations or the idle "pulse" animation depending on the current state of the navbar.
